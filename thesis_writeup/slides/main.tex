\section{Motivation \& Problem}

\begin{frame}{Why another truss-optimisation thesis?}
  \begin{columns}[t,onlytextwidth]
    \column{0.58\textwidth}
      \begin{itemize}\itemsep0.4em
        \item India builds thousands of industrial sheds,
              transmission towers, and highway overpasses each
              year under IS~800:2007.
        \item Classical trial-and-check design typically carries
              \textbf{10--30\,\% surplus steel} versus the
              theoretical feasible optimum.
        \item Steel at $\sim\!\num{1.9}\,\text{kg}~\text{CO}_2/\text{kg}$
              $\Rightarrow$ a \num{15}\,\% cut is sector-level
              impact.
        \item Existing optimisation tools exist, but sit on the
              researcher's bench --- a full GA run is
              40~min/benchmark, too slow for iterative design.
      \end{itemize}
    \column{0.42\textwidth}
      \centering
      \includegraphics[width=\linewidth]{fig_1_1_motivation.png}
      {\scriptsize Design-gap between classical and optimal.}
  \end{columns}
\end{frame}

\begin{frame}{Research question}
  \begin{block}{Central claim}
    Seeding GA / PSO / NSGA-II populations with designs
    generated by a modern \textbf{large language model}
    (Claude Opus) converges faster than random initialisation on
    standard truss weight-minimisation benchmarks, while
    remaining compliant with IS~800:2007 limit-state provisions.
  \end{block}
  \begin{block}{Five research objectives}
    \begin{enumerate}\small
      \item Reproduce classical benchmark optima to within $2\,\%$.
      \item Return NSGA-II Pareto fronts with $\ge\!20$
            non-dominated points.
      \item Achieve $\ge\!50\times$ speedup via neural surrogate
            at $R^2>\num{0.98}$.
      \item Train PPO agent within $5\,\%$ of best classical
            solver.
      \item LLM warm-start reduces generations-to-convergence by
            $\ge\!20\,\%$.
    \end{enumerate}
  \end{block}
\end{frame}

  \small
  \begin{description}\itemsep0.25em
    \item[\textbf{C1}] End-to-end framework integrating FEM,

      design-aid: full pipeline $<10$~s on laptop CPU;
      every reported number reproducible against tag
      \texttt{phase-10-complete} without an API key.
  \end{description}
\end{frame}

\section{Problem definition}

\begin{frame}{Weight minimisation under codal constraints}
  \begin{equation*}
    \min_{A\in\mathcal{A}}\; W(A) = \sum_{i=1}^{N} \rho\,L_i\,A_i
    \quad\text{s.t.}\quad
    \begin{cases}
      |\sigma_i(A)| \le \sigma_{\text{all},i}(A) & \text{IS 800 \S\S 6, 7}\\
      |\delta_j(A)| \le \delta_{\max} & \text{IS 800 \S 5.6.1}\\
      \lambda_i(A) \le 180 & \text{IS 800 \S 3.8}\\
      A_i \in [A_{\min},A_{\max}] &
    \end{cases}
  \end{equation*}
  \vspace{-0.2em}
  \begin{block}{Four canonical benchmarks}
    \begin{tabular}{@{}lllr@{}}
      \toprule
      Benchmark & Type & DoF & Lit.\ optimum\\
      \midrule
      10-bar planar & sizing & 10 & \num{5060.85}~lb\\
      25-bar spatial & sizing (grouped) & 8 & \num{545.16}~lb\\
      72-bar spatial tower & sizing (grouped) & 16 & \num{379.62}~lb$^\dagger$\\
      200-bar planar tower & sizing (grouped) & 29 & \num{25445}~lb\\
      \bottomrule
    \end{tabular}
    \par\vspace{0.3em}{\scriptsize $^\dagger$ shown in this thesis
    to be infeasible under hard constraints.}
  \end{block}
\end{frame}

\section{Methodology}

\begin{frame}{FEM kernel $+$ IS~800 checks}
  \begin{columns}[t,onlytextwidth]
    \column{0.48\textwidth}
      \textbf{FEM (linear elastic trusses)}
      \begin{itemize}\small
        \item Global stiffness assembly $K = \sum_e B_e^\top k_e B_e$.
        \item Partition method applied to enforce supports.
        \item Solve $K_{ff}\,u_f = F_f$ via dense \texttt{numpy.linalg.solve}.
        \item Axial stress recovery $\sigma_e = (E_e/L_e)\,[-c\ -s\ c\ s]\,u_e$.
      \end{itemize}
      \vspace{0.3em}
      \includegraphics[width=\linewidth]{fig_3_1_truss_element.png}
    \column{0.48\textwidth}
      \textbf{IS~800:2007 compliance} (Appendix~B)
      \begin{itemize}\small
        \item Tension yield \S6.2: $T_{dg}=A_g f_y/\gamma_{m0}$.
        \item Tension rupture \S6.3: $T_{dn}=0.9 A_n f_u/\gamma_{m1}$.
        \item Compression buckling \S7.1 via Perry-Robertson,
              buckling curve~$a$.
        \item Slenderness $\lambda\le 180$ \S3.8.
        \item Serviceability $\delta\le L/325$ \S5.6.1.
      \end{itemize}
      \vspace{0.3em}
      \includegraphics[width=\linewidth]{fig_3_8_is800_checks.png}
  \end{columns}
\end{frame}

\begin{frame}{Three evolutionary baselines}

  \begin{columns}[t,onlytextwidth]

      \textbf{GA} (pymoo)\\
      pop~100, gens~500\\

\end{frame}

\begin{frame}{Neural surrogate (C2)}
  \begin{columns}[t,onlytextwidth]
    \column{0.5\textwidth}
      \textbf{Architecture}
      \begin{itemize}\small
        \item MLP $[n_A] \to 256 \to 128 \to 64 \to 3$.
        \item Multi-head output: weight, max-stress,
              max-displacement.
        \item Trained on \num{10000} LHS samples, Adam
              lr=$\num{1e-3}$, 200 epochs, dropout $p=0.2$.
      \end{itemize}
      \textbf{Use}
      \begin{itemize}\small
        \item Drop-in FEM replacement in pymoo inner loop.
        \item Hybrid: surrogate screen $\to$ FEM fallback when
              $|\delta_{\text{surr}}|>\delta_{\max}$.
      \end{itemize}
    \column{0.5\textwidth}
      \includegraphics[width=\linewidth]{fig_3_5_surrogate_arch.png}
      \vspace{0.4em}
      \begin{block}{Headline}
        $R^2=\num{0.9993}$ on weight, $\num{132}\times$
        wall-clock speedup, 10-bar.
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{RL layer (PPO) and LLM warm-start}
  \begin{columns}[t,onlytextwidth]
    \column{0.5\textwidth}
      \textbf{PPO MDP}
      \begin{itemize}\small
        \item State: current $A\in\mathbb{R}^{n_A}$, $\sigma_{\max}$, $\delta_{\max}$.
        \item Action: multiplicative adjustment $\in[0.5,2.0]$.
        \item Reward: $-W - \lambda\cdot\text{infeas}$.
      \end{itemize}
      \includegraphics[width=\linewidth]{fig_3_6_rl_mdp.png}
    \column{0.5\textwidth}
      \textbf{LLM warm-start pipeline}
      \begin{itemize}\small
        \item Structured prompt $\Rightarrow$ Claude Opus
              returns $k{=}8$ candidate design vectors +
              reasoning.
        \item Injected as first-population seeds (the rest are
              LHS).
        \item All responses cached under
              \texttt{results/llm\_cache/} for reproducibility.
      \end{itemize}
      \includegraphics[width=\linewidth]{fig_3_7_llm_pipeline.png}
  \end{columns}
\end{frame}

\section{Conclusions \& future work}

\begin{frame}{What the thesis adds}
  \small
  \begin{block}{Concrete numbers}
    \begin{itemize}
      \item $\num{0.004}\,\%$ 10-bar error $\Rightarrow$ stack
            validated end-to-end.
      \item $\num{132}\times$ speedup at $R^2=\num{0.9993}$.
      \item $\num{36.8}\,\%$ LLM-warm-start generation reduction
            ($p=\num{0.0046}$) on 10-bar.
      \item 72-bar literature optimum shown infeasible by
            \num{22}--\num{62}\,\%.
    \end{itemize}
  \end{block}
  \begin{block}{Limitations}
    Linear-elastic FEM, sizing-only (no topology), small seed
    counts on 25/72-bar NSGA-II, single buckling curve~$a$, LLM
    not fine-tuned.
  \end{block}
  \begin{block}{Future work (F1--F6)}
    Non-linear / dynamic FEM $\to$ IS~1893 seismic;
    topology via ground-structure; GNN surrogate for
    variable-topology; cross-instance RL transfer; fine-tuned
    domain-specific LLM; PGCIL transmission-tower validation.
  \end{block}
\end{frame}

\begin{frame}{Practical implication for Indian practice}
  \begin{itemize}
    \item Hybrid surrogate/FEM collapses the
          \num{30}--\num{40}-minute-per-run cost of classical
          optimisation into $<20$~s --- \textbf{interactive
          parametric studies} in a design-review meeting.
    \item 72-bar rigorous-feasibility finding is a
          first-order warning for Indian transmission-tower
          practice --- commercial soft-penalty defaults may
          carry \num{22}--\num{62}\,\% infeasibility.
    \item Streamlit UI exposes the whole stack without
          Python literacy --- consumable by a junior design
          engineer at a consulting firm.
  \end{itemize}
\end{frame}
